class Solution {
    public int[] twoSum(int[] numbers, int target) {
          int i=0,j=numbers.length-1;

          while(i<j){
            int check= numbers[i]+numbers[j];
            if(check==target){return new int[]{i+1,j+1};}

            if(check>target){
                j--;
            }

            if(check<target){
                i++;
            }
          }

         return new int[]{1,2}; 
          
    }
}
