import java.util.ArrayDeque;
import java.util.Deque;

class Solution {

    public int minOperations(int[] nums) {
       
        Deque<Integer> stack = new ArrayDeque<>();
        
        int operations = 0;

        for (int value : nums) {

            
            if (value == 0) {
                stack.clear();
                continue;
            }

           
            while (!stack.isEmpty() && stack.peekLast() > value) {
                stack.pollLast();
            }

           
            if (!stack.isEmpty() && stack.peekLast() == value) {
                continue;
            }

            
            stack.addLast(value);
            operations++;
        }

        return operations;
    }
}
