class Solution {

    static{
        for(int i=0;i<1000;i++){
            findFinalValue(new int[]{2,3,4,43,3}, 2);
            findFinalValue(new int[]{2,3,4,43,3}, 5);
        }
    }


    static public int findFinalValue(int[] nums, int original) {
        

        Arrays.sort(nums);

        for(int ele:nums){
            if(ele>original){break;}
            else if(ele==original){
                original*=2;
            }
            
        }   

        return original;


    }
}
