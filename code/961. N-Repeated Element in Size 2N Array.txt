class Solution {
    public int repeatedNTimes(int[] nums) {
        int max=0;
        for(int a:nums)max=Math.max(a,max);
        int[] fre = new int[max+1];
        for(int a:nums)fre[a]++;
        int n=nums.length/2;
        for(int i=0;i<fre.length;i++){
            if(fre[i]==n)return i;
        }
        return 0;
    }
}
