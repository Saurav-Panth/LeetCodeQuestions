class Solution {

    static{
        for(int i=0;i<1001;i++){
            kLengthApart(new int[]{0,1,1,1,0}, 4);
            kLengthApart(new int[]{0,1,1,1,0}, 1);
        }
    }

    public static boolean kLengthApart(int[] nums, int k) {
        int count=k;
        for(int i =0;i<nums.length;i++){
            if(nums[i]==0){count++;}
            if(nums[i]==1){
                if(count<k){return false;}
                count=0;
            }
        }
        return true;
    }
}
