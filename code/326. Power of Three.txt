class Solution {
    public boolean isPowerOfThree(int n) {
        if(n==1)return true;
        return check(n,1);
    }

   static boolean check(int n,int m){
        double p =Math.pow(3,m);
        // System.out.println(p+" "+m);
        if(p==n)return true;
        if(p>n)return false;
        return check(n,++m);
    }

    

}
