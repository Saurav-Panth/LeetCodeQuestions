class Solution {

    static {
        for(int i=0;i<1000;i++){
            getDescentPeriods(new int[]{2,1,2,2});
            getDescentPeriods(new int[]{2,3,4,5,66});
        }
    }

   static public long getDescentPeriods(int[] prices) {
        long ans =1,count=1;

        for(int i=1;i<prices.length;i++){
         
            if(prices[i]+1==prices[i-1])count++;
            else count=1;
            ans+=count;
            
        }
        return ans;

    }
}
