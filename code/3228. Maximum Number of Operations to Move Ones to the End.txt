class Solution {

    static{
        for(int i=0;i<1000;i++){
            maxOperations("dssdas");
            maxOperations("dssdadws");
        }
    }


    static public int maxOperations(String s) {
    
      int count=0,one=0;
    
    

      for(int i =0;i<s.length()-1;i++){
            if(s.charAt(i)=='1')one++;
            if(s.charAt(i)=='1' && s.charAt(i+1)=='0'){count+=one;}
      }


      return count;
    }
}
