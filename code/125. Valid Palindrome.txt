class Solution {
    public boolean isPalindrome(String s) {
        String str = s.toLowerCase();
        int i = 0, j = str.length() - 1;
        while (i < j) {
            char a = str.charAt(i);
            char b = str.charAt(j);

            if (!Character.isLetterOrDigit(a)) { i++; continue; }
            if (!Character.isLetterOrDigit(b)) { j--; continue; }

            if (a != b) return false;

            i++;
            j--;
        }
        return true;
    }
}
