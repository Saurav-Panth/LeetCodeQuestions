class Solution {

    static{
        for(int i=0;i<1001;i++){
            minOperations(new int[]{4,5,2}, 3);
            minOperations(new int[]{1,2,3}, 4);
        }
    }


    static public int minOperations(int[] nums, int k) {
        int sum =0;
        for(int a:nums){sum+=a;}

        return sum-(k*(sum/k));


    }
}
